import json
import ssl
import time
import base64
import urllib.request

MAX_WAIT_SECONDS = 300
SLEEP_INTERVAL = 10


def handler(context, inputs):

    vc = context.getSecret("VCENTER_SERVER")
    user = context.getSecret("VCENTER_USER")
    password = context.getSecret("VCENTER_PASSWORD")

    vm_id = inputs["resourceIds"][0]

    context.logger.info("Processing VM: " + vm_id)

    session = vc_login(vc, user, password)

    ip = wait_for_ip(vc, session, vm_id, context)

    if not ip:
        context.logger.warn("No DHCP IP discovered within timeout")
        return

    rename_vm(vc, session, vm_id, ip, context)


def vc_login(vc, user, password):

    url = "https://" + vc + "/rest/com/vmware/cis/session"

    pair = user + ":" + password
    token = base64.b64encode(pair.encode()).decode()

    headers = {
        "Authorization": "Basic " + token
    }

    req = urllib.request.Request(url, method="POST", headers=headers)

    ctx = ssl._create_unverified_context()

    with urllib.request.urlopen(req, context=ctx) as r:
        body = json.loads(r.read().decode())

    return body["value"]


def wait_for_ip(vc, session, vm_id, context):

    headers = {
        "vmware-api-session-id": session
    }

    url = "https://" + vc + "/rest/vcenter/vm/" + vm_id + "/guest/networking/interfaces"

    elapsed = 0

    ctx = ssl._create_unverified_context()

    while elapsed < MAX_WAIT_SECONDS:

        try:

            req = urllib.request.Request(url, headers=headers)

            with urllib.request.urlopen(req, context=ctx) as r:
                data = json.loads(r.read().decode())

            ip = extract_ipv4(data.get("value", []))

            if ip:
                context.logger.info("Discovered DHCP IP: " + ip)
                return ip

        except Exception as e:
            context.logger.warn(str(e))

        context.logger.info("Waiting for DHCP IP...")

        time.sleep(SLEEP_INTERVAL)
        elapsed += SLEEP_INTERVAL

    return None


def extract_ipv4(interfaces):

    for iface in interfaces:

        ipblock = iface.get("ip")

        if not ipblock:
            continue

        for entry in ipblock.get("ip_addresses", []):

            addr = entry.get("ip_address")

            if not addr:
                continue

            if "." not in addr:
                continue

            if addr.startswith("169.254"):
                continue

            if addr.startswith("127."):
                continue

            return addr

    return None


def rename_vm(vc, session, vm_id, ip, context):

    ctx = ssl._create_unverified_context()

    headers = {
        "vmware-api-session-id": session
    }

    vm_url = "https://" + vc + "/rest/vcenter/vm/" + vm_id

    req = urllib.request.Request(vm_url, headers=headers)

    with urllib.request.urlopen(req, context=ctx) as r:
        data = json.loads(r.read().decode())

    current_name = data["value"]["name"]

    if "(" in current_name:
        context.logger.info("VM already renamed")
        return

    new_name = current_name + " (" + ip + ")"

    payload = json.dumps({
        "spec": {
            "name": new_name
        }
    }).encode()

    headers["Content-Type"] = "application/json"

    req = urllib.request.Request(
        vm_url,
        data=payload,
        headers=headers,
        method="PATCH"
    )

    urllib.request.urlopen(req, context=ctx)

    context.logger.info("VM renamed to: " + new_name)
