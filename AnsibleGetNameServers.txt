import requests
import time
import ssl

requests.packages.urllib3.disable_warnings()

MAX_WAIT_SECONDS = 300
SLEEP_INTERVAL = 10


def handler(context, inputs):

    vc = context.getSecret("VCENTER_SERVER")
    user = context.getSecret("VCENTER_USER")
    password = context.getSecret("VCENTER_PASSWORD")

    vm_id = inputs["resourceIds"][0]

    context.logger.info(f"Processing VM {vm_id}")

    session = vc_login(vc, user, password)

    ip = wait_for_ip(vc, session, vm_id, context)

    if not ip:
        context.logger.warn("No DHCP IP discovered within timeout")
        return

    rename_vm(vc, session, vm_id, ip, context)


def vc_login(vc, user, password):

    url = f"https://{vc}/rest/com/vmware/cis/session"

    r = requests.post(
        url,
        auth=(user, password),
        verify=False
    )

    r.raise_for_status()

    return r.json()["value"]


def wait_for_ip(vc, session, vm_id, context):

    headers = {
        "vmware-api-session-id": session
    }

    url = f"https://{vc}/rest/vcenter/vm/{vm_id}/guest/networking/interfaces"

    elapsed = 0

    while elapsed < MAX_WAIT_SECONDS:

        try:
            r = requests.get(url, headers=headers, verify=False)

            if r.status_code == 200:

                data = r.json()["value"]

                ip = extract_ipv4(data)

                if ip:
                    context.logger.info(f"Discovered DHCP IP: {ip}")
                    return ip

        except Exception as e:
            context.logger.warn(str(e))

        context.logger.info("Waiting for DHCP IP...")
        time.sleep(SLEEP_INTERVAL)
        elapsed += SLEEP_INTERVAL

    return None


def extract_ipv4(interfaces):

    for iface in interfaces:

        for ip in iface.get("ip", {}).get("ip_addresses", []):

            addr = ip.get("ip_address")

            if not addr:
                continue

            if "." not in addr:
                continue

            if addr.startswith("169.254"):
                continue

            if addr.startswith("127."):
                continue

            return addr

    return None


def rename_vm(vc, session, vm_id, ip, context):

    headers = {
        "vmware-api-session-id": session
    }

    vm_url = f"https://{vc}/rest/vcenter/vm/{vm_id}"

    r = requests.get(vm_url, headers=headers, verify=False)

    r.raise_for_status()

    current_name = r.json()["value"]["name"]

    new_name = f"{current_name} ({ip})"

    payload = {
        "spec": {
            "name": new_name
        }
    }

    context.logger.info(f"Renaming VM to {new_name}")

    r = requests.patch(
        vm_url,
        json=payload,
        headers=headers,
        verify=False
    )

    r.raise_for_status()

    context.logger.info("Rename successful")
